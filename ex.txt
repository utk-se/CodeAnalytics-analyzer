#include <stdio.h>
int main() {
    int low, high, i, flag, nr;
    printf("Enter two numbers(intervals): ");
    scanf("%d %d", &low, &high);
    printf("Prime numbers between %d and %d are: ", low, high);

    while (low < high) {
        flag = 0; // yo

        // if low is a non-prime number, flag will be 1
        for (i = 2; i <= low / 2; ++i) {
            if (low % i == 0) {
                flag = 1;
                break;
            }
        }

        if (flag == 0)
            nr = printf("%d ", low);
        ++low;
    }

    return 0;
}
